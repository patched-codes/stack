# SocialMediaIcons

A component that renders social media icons with links.

## Example

```tsx
import { SocialMediaIcons } from '@stackframe/stack';

const MyComponent = () => {
  return (
    <div>
      <h1>Connect with me</h1>
      <SocialMediaIcons />
    </div>
  );
};
```
